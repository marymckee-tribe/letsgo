# AI Chat Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Wire the bottom chat bar to Claude Haiku via `streamText`, show streaming responses inline above the input, and dispatch store actions (add task/event/grocery, complete task) from server-side tool calls.

**Architecture:** New `POST /api/chat` route uses `streamText` with four server-side tools (execute functions return `{ success: true }`). The command bar replaces its manual submit handler with `useChat` from `ai/react`. A `useEffect` watches the messages array for completed tool invocations and dispatches directly to the store. The inline panel grows above the input when messages exist.

**Tech Stack:** `ai@^6` (Vercel AI SDK), `@ai-sdk/anthropic`, `ai/react` `useChat`, `streamText`, `tool`, `zod`

```ts
// src/app/api/chat/route.ts
export const runtime = 'nodejs'
export const maxDuration = 30

import { anthropic } from '@ai-sdk/anthropic'
import { streamText, tool } from 'ai'
import { z } from 'zod'
import { NextResponse } from 'next/server'

export async function POST(req: Request) {
  try {
    const authHeader = req.headers.get('authorization')
    if (!authHeader) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

    const { messages, context } = await req.json()

    const result = streamText({
      model: anthropic('claude-haiku-4-5-20251001'),
      system: buildSystemPrompt(context),
      messages,
      maxSteps: 3,
      tools: {
        add_task: tool({
          description: "Add a task to the user's planner",
          parameters: z.object({
            title: z.string(),
            context: z.enum(['FAMILY', 'WORK', 'HOUSEHOLD', 'PERSONAL', 'ERRANDS']),
            due: z.enum(['today', 'this-week', 'someday']),
            who: z.string().optional(),
          }),
          execute: async (args) => ({ success: true, ...args }),
        }),
        add_event: tool({
          description: 'Add an event to the calendar',
          parameters: z.object({
            title: z.string(),
            date: z.number().describe('Day of month (1-31)'),
            time: z.string().describe('HH:MM format'),
          }),
          execute: async (args) => ({ success: true, ...args }),
        }),
        add_grocery: tool({
          description: 'Add an item to the grocery list',
          parameters: z.object({
            name: z.string(),
            quantity: z.string().optional(),
          }),
          execute: async (args) => ({ success: true, ...args }),
        }),
        complete_task: tool({
          description: 'Mark a task as complete by its ID',
          parameters: z.object({
            id: z.string().describe('The exact task ID from context'),
          }),
          execute: async (args) => ({ success: true, ...args }),
        }),
      },
    })

    return result.toDataStreamResponse()
  } catch (error: unknown) {
    const message = error instanceof Error ? error.message : String(error)
    return NextResponse.json({ error: message }, { status: 500 })
  }
}

function buildSystemPrompt(context: any): string {
  const dateStr = new Date().toLocaleDateString('en-US', {
    weekday: 'long', year: 'numeric', month: 'long', day: 'numeric',
  })

  const lines: string[] = [
    `You are a Chief of Staff AI for this family. Today is ${dateStr}.`,
    'You have full context of their tasks, calendar, emails, and family profiles.',
    'Be concise and action-oriented. When asked to do something, use the available tools.',
    'After using a tool, confirm what you did in one sentence.',
    '',
  ]

  const { tasks = [], events = [], emails = [], profiles = [], groceries = [] } = context ?? {}

  const activeTasks = tasks.filter((t: any) => !t.completed)
  if (activeTasks.length > 0) {
    lines.push(`TASKS (${activeTasks.length} active):`)
    activeTasks.forEach((t: any) => {
      lines.push(`- id:${t.id} "${t.title}" [${t.context}${t.due ? `, ${t.due}` : ''}${t.who ? `, ${t.who}` : ''}]`)
    })
    lines.push('')
  }

  if (events.length > 0) {
    lines.push(`CALENDAR (${events.length} events):`)
    events.forEach((e: any) => lines.push(`- ${e.title}, date ${e.date}, ${e.time}`))
    lines.push('')
  }

  if (emails.length > 0) {
    lines.push(`EMAILS (${emails.length} unread):`)
    emails.forEach((e: any) => lines.push(`- From: ${e.sender ?? 'Unknown'} | "${e.subject}": ${e.snippet}`))
    lines.push('')
  }

  if (profiles.length > 0) {
    lines.push('PROFILES:')
    profiles.forEach((p: any) => {
      const parts = [`${p.name} (${p.type})`]
      if (p.medicalFlags?.length) parts.push(`Medical: ${p.medicalFlags.join(', ')}`)
      if (p.dietary?.length) parts.push(`Dietary: ${p.dietary.join(', ')}`)
      if (p.routines?.length) {
        parts.push(`Routines: ${p.routines.map((r: any) => `${r.activity} ${r.day} ${r.time}`).join(', ')}`)
      }
      lines.push(`- ${parts.join(' | ')}`)
    })
    lines.push('')
  }

  const unchecked = groceries.filter((g: any) => !g.checked)
  if (unchecked.length > 0) {
    lines.push(`GROCERIES (${unchecked.length} needed):`)
    unchecked.forEach((g: any) => lines.push(`- ${g.name}${g.quantity ? ` (${g.quantity})` : ''}`))
    lines.push('')
  }

  return lines.join('\n')
}
```

```tsx
// src/components/ai-command-bar.tsx
"use client"

import { useRef, useEffect } from "react"
import { usePathname } from "next/navigation"
import { useChat } from "ai/react"
import { useHub } from "@/lib/store"
import { useTerminal } from "@/lib/terminal-context"
import { useAuth } from "@/lib/auth-provider"

const QUICK_PROMPTS = [
  "What needs attention today?",
  "Schedule Annie's vaccine",
  "What's on this week?",
]

export function AiCommandBar() {
  const pathname = usePathname()
  const { tasks, events, emails, profiles, groceries, addTask, addEvent, addGrocery, toggleTask } = useHub()
  const { registerSeed } = useTerminal()
  const { accessToken } = useAuth()
  const dispatchedRef = useRef<Set<string>>(new Set())
  const panelRef = useRef<HTMLDivElement>(null)

  const {
    messages,
    input,
    handleInputChange,
    handleSubmit: chatSubmit,
    isLoading,
    setMessages,
    setInput,
    error,
  } = useChat({ api: "/api/chat" })

  // Wire registerSeed so other components can pre-fill the input
  useEffect(() => {
    registerSeed((content: string) => setInput(content))
  }, [registerSeed, setInput])

  // Scroll panel to bottom when new messages arrive
  useEffect(() => {
    if (panelRef.current) {
      panelRef.current.scrollTop = panelRef.current.scrollHeight
    }
  }, [messages])

  // Dispatch store actions from completed server-side tool invocations
  useEffect(() => {
    messages.forEach(msg => {
      if (msg.role !== "assistant") return
      ;((msg as any).toolInvocations ?? []).forEach((inv: any) => {
        if (inv.state !== "result") return
        if (dispatchedRef.current.has(inv.toolCallId)) return
        dispatchedRef.current.add(inv.toolCallId)

        if (inv.toolName === "add_task") {
          addTask({ id: crypto.randomUUID(), completed: false, ...inv.args })
        } else if (inv.toolName === "add_event") {
          addEvent({ id: crypto.randomUUID(), ...inv.args })
        } else if (inv.toolName === "add_grocery") {
          addGrocery({ id: crypto.randomUUID(), ...inv.args })
        } else if (inv.toolName === "complete_task") {
          toggleTask(inv.args.id)
        }
      })
    })
  }, [messages, addTask, addEvent, addGrocery, toggleTask])

  if (pathname === "/login") return null

  const hasMessages = messages.length > 0

  const handleFormSubmit = (e: React.FormEvent<HTMLFormElement>) => {
    chatSubmit(e, {
      headers: { Authorization: `Bearer ${accessToken ?? ""}` },
      body: {
        context: {
          tasks,
          events,
          emails: emails.map(em => ({ subject: em.subject, sender: em.sender, snippet: em.snippet })),
          profiles,
          groceries,
        },
      },
    })
  }

  const handleClear = () => {
    setMessages([])
    dispatchedRef.current.clear()
  }

  // Is the last assistant message still being streamed (empty content)?
  const lastMsg = messages[messages.length - 1]
  const showThinking = isLoading && lastMsg?.role === "user"

  return (
    <div className="shrink-0 border-t border-black/10 bg-white" style={{ fontFamily: "var(--font-dm-sans, system-ui)" }}>
      {/* Status line */}
      <div className="flex items-center gap-4 px-12 pt-3">
        <div className="flex items-center gap-1.5">
          <div className="w-1.5 h-1.5 rounded-full bg-[#4285f4] animate-pulse" />
          <span className="text-[9px] uppercase tracking-[0.2em] font-bold text-[#4285f4]">AI Active</span>
        </div>
        <span className="text-black/15">·</span>
        <span className="text-[10px] text-black/25">Monitoring Gmail · Slack · Life Graph</span>
      </div>

      {/* Conversation panel — grows above the input */}
      {hasMessages && (
        <div
          ref={panelRef}
          className="mx-12 mt-3 mb-2 max-h-72 overflow-y-auto border border-black/8 p-4"
          style={{ scrollbarWidth: "none" }}
        >
          <div className="flex justify-end mb-2">
            <button
              onClick={handleClear}
              className="text-[9px] uppercase tracking-[0.2em] text-black/25 hover:text-black transition-colors"
            >
              × Clear
            </button>
          </div>
          <div className="space-y-3">
            {messages.map(msg => (
              <div key={msg.id} className={`flex ${msg.role === "user" ? "justify-end" : "justify-start"}`}>
                <p
                  className={`max-w-[80%] text-sm leading-relaxed ${
                    msg.role === "user"
                      ? "text-black font-medium"
                      : "font-serif italic text-black/70"
                  }`}
                >
                  {msg.content}
                </p>
              </div>
            ))}
            {showThinking && (
              <div className="flex justify-start">
                <p className="text-sm font-serif italic text-black/30">Thinking...</p>
              </div>
            )}
            {error && (
              <p className="text-xs text-red-500 text-center">Something went wrong. Try again.</p>
            )}
          </div>
        </div>
      )}

      {/* Input row */}
      <form onSubmit={handleFormSubmit} className="flex items-center gap-4 px-12 py-3">
        <span className="text-[10px] uppercase tracking-[0.25em] text-[#4285f4] font-bold shrink-0">Chat</span>
        <input
          type="text"
          value={input}
          onChange={handleInputChange}
          placeholder="What needs to happen today..."
          className="flex-1 bg-transparent text-sm text-black placeholder:text-black/20 outline-none"
        />
        {input ? (
          <button
            type="submit"
            className="text-[10px] uppercase tracking-[0.2em] font-bold text-black/40 hover:text-black transition-colors shrink-0"
          >
            ↵ Send
          </button>
        ) : !hasMessages ? (
          <div className="flex items-center gap-2 shrink-0">
            {QUICK_PROMPTS.map(prompt => (
              <button
                key={prompt}
                type="button"
                onClick={() => setInput(prompt)}
                className="text-[10px] text-black/25 border border-black/8 px-3 py-1 hover:border-black/25 hover:text-black/50 transition-colors whitespace-nowrap hidden lg:block"
              >
                {prompt}
              </button>
            ))}
          </div>
        ) : null}
      </form>
    </div>
  )
}
```
